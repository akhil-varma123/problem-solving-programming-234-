#include <stdio.h>


int main()
 {
 
   int a,b,c;
   printf("enter the values of three sides:");
 
   scanf("%d%d%d",&a,&b,&c);

    if(a==b&&b==c)

   {
  
     printf("equilateral triangle");

   }
   
    else if(a==b&&b!=c)

   {
      
    printf("isosceles triangle");

   }
   
  else
  
 {
        
  printf("scalene");
 
   }
     
  return 0;
 
}
